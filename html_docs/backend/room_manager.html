247    def check_availability_for_dates(self, room_type, check_in, check_out, calendar_head):
248        """
249        Check date-based availability for a room type using the calendar system.
250        Args:
251            room_type (str): Type of room.
252            check_in (tuple[int, int]): (month, day)
253            check_out (tuple[int, int]): (month, day)
254            calendar_head: Calendar node reference.
255        Returns:
256            tuple(bool, str): Availability status and explanation.
257        """
258        rooms_of_type = self.get_rooms_by_type(room_type)
259        total_rooms = len(rooms_of_type)
260
261        month, day = check_in
262        end_month, end_day = check_out
263
264        room_type_dict = {
265            "name": room_type,
266            "quantity": total_rooms
267        }
268
269        while True:
270            booked = get_booked_quantity(month, day, room_type_dict, calendar_head)
271            if booked >= total_rooms:
272                return False, f"No {room_type} available on {month}/{day}"
273
274            if (month, day) == (end_month, end_day):
275                break
276
277            day += 1
278            if day > 30:
279                day, month = 1, month + 1
280
281        return True, f"{total_rooms - booked} {room_type}(s) available"

322    def search_rooms(self, min_price=None, max_price=None, min_guests=None, available_only=True, clean_only=False):
323        """
324        Search rooms with multiple filters.
325        Args:
326            min_price (float | None): Minimum room price.
327            max_price (float | None): Maximum room price.
328            min_guests (int | None): Minimum guest capacity.
329            available_only (bool): Only include available rooms.
330            clean_only (bool): Only include clean rooms.
331        Returns:
332            list[Room]: Filtered room list.
333        """
334        results = self.hotel.rooms
335
336        if available_only:
337            results = [room for room in results if room.is_available]
338
339        if clean_only:
340            results = [room for room in results if self.room_cleanliness.get(room.room_id) == "clean"]
341
342        if min_price is not None:
343            results = [room for room in results if room.price >= min_price]
344
345        if max_price is not None:
346            results = [room for room in results if room.price <= max_price]
347
348        if min_guests is not None:
349            results = [room for room in results if room.max_guests >= min_guests]
350
351        return results

249        """
250        Check date-based availability for a room type using the calendar system.
251        Args:
252            room_type (str): Type of room.
253            check_in (tuple[int, int]): (month, day)
254            check_out (tuple[int, int]): (month, day)
255            calendar_head: Calendar node reference.
256        Returns:
257            tuple(bool, str): Availability status and explanation.
258        """
259        rooms_of_type = self.get_rooms_by_type(room_type)
260        total_rooms = len(rooms_of_type)
261
262        month, day = check_in
263        end_month, end_day = check_out
264
265        room_type_dict = {
266            "name": room_type,
267            "quantity": total_rooms
268        }
269
270        while True:

272            if booked >= total_rooms:
273                return False, f"No {room_type} available on {month}/{day}"
274
275            if (month, day) == (end_month, end_day):
276                break
277
278            day += 1
279            if day > 30:
280                day, month = 1, month + 1
281
282        return True, f"{total_rooms - booked} {room_type}(s) available"
